```python
        viewport={"width": 1280, "height": 800},

    )
    
    # ‡πÄ‡∏û‡∏¥‡πà‡∏° cookies

        "name": "session_id", 
        "value": "test123", 
        "domain": "example.com",
        "path": "/",
    }])
    
    page = context.new_page()
    
    # ‡πÄ‡∏õ‡∏¥‡∏î‡πÄ‡∏ß‡πá‡∏ö‡πÑ‡∏ã‡∏ï‡πå
    page.goto("https://example.com/login")

    # ‡∏Å‡∏£‡∏≠‡∏Å‡∏ü‡∏≠‡∏£‡πå‡∏°‡πÅ‡∏ö‡∏ö‡πÄ‡∏´‡∏°‡∏∑‡∏≠‡∏ô‡∏°‡∏ô‡∏∏‡∏©‡∏¢‡πå
```

## ‡∏Å‡∏≤‡∏£‡πÉ‡∏ä‡πâ‡∏á‡∏≤‡∏ô LangGraph (AI Workflow)

LangGraph ‡∏ä‡πà‡∏ß‡∏¢‡πÉ‡∏´‡πâ‡∏Ñ‡∏∏‡∏ì‡∏™‡∏≤‡∏°‡∏≤‡∏£‡∏ñ‡∏™‡∏£‡πâ‡∏≤‡∏á workflow ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö AI agent ‡∏ó‡∏µ‡πà‡∏°‡∏µ‡∏Ñ‡∏ß‡∏≤‡∏°‡∏ã‡∏±‡∏ö‡∏ã‡πâ‡∏≠‡∏ô‡πÑ‡∏î‡πâ ‡πÇ‡∏î‡∏¢‡∏°‡∏µ‡∏£‡∏∞‡∏ö‡∏ö retry ‡πÅ‡∏•‡∏∞ state management

### ‡∏ï‡∏±‡∏ß‡∏≠‡∏¢‡πà‡∏≤‡∏á‡∏Å‡∏≤‡∏£‡πÉ‡∏ä‡πâ‡∏á‡∏≤‡∏ô:

```python

from langchain_openai import ChatOpenAI
from langgraph.graph import END, StateGraph
import os

# ‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤ API key
os.environ["OPENAI_API_KEY"] = "your-api-key-here"

# ‡∏ô‡∏¥‡∏¢‡∏≤‡∏° state
class AgentState(dict):
    """State for the agent workflow"""
    messages: list
    attempts: int
    status: str
    card_bin: str

# ‡∏™‡∏£‡πâ‡∏≤‡∏á workflow
workflow = StateGraph(AgentState)

# ‡∏ô‡∏¥‡∏¢‡∏≤‡∏° node functions
def analyze_gateway(state):
    """‡∏ß‡∏¥‡πÄ‡∏Ñ‡∏£‡∏≤‡∏∞‡∏´‡πå‡∏£‡∏∞‡∏ö‡∏ö payment gateway"""
    # ‡πÇ‡∏Ñ‡πâ‡∏î‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö‡∏ß‡∏¥‡πÄ‡∏Ñ‡∏£‡∏≤‡∏∞‡∏´‡πå gateway
    return {"status": "analyzed"}

def select_card(state):
    """‡πÄ‡∏•‡∏∑‡∏≠‡∏Å‡∏ö‡∏±‡∏ï‡∏£‡∏ó‡∏µ‡πà‡πÄ‡∏´‡∏°‡∏≤‡∏∞‡∏™‡∏°"""
    # ‡πÇ‡∏Ñ‡πâ‡∏î‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö‡πÄ‡∏•‡∏∑‡∏≠‡∏Å‡∏ö‡∏±‡∏ï‡∏£
    return {"card_bin": "123456"}

def process_payment(state):
    """‡∏î‡∏≥‡πÄ‡∏ô‡∏¥‡∏ô‡∏Å‡∏≤‡∏£‡∏ä‡∏≥‡∏£‡∏∞‡πÄ‡∏á‡∏¥‡∏ô"""
    # ‡πÇ‡∏Ñ‡πâ‡∏î‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö‡∏Å‡∏≤‡∏£‡∏ä‡∏≥‡∏£‡∏∞‡πÄ‡∏á‡∏¥‡∏ô
    success = state["attempts"] < 2  # ‡∏à‡∏≥‡∏•‡∏≠‡∏á‡∏Å‡∏≤‡∏£‡∏•‡πâ‡∏°‡πÄ‡∏´‡∏•‡∏ß‡πÉ‡∏ô‡∏Ñ‡∏£‡∏±‡πâ‡∏á‡∏ó‡∏µ‡πà 3
    if success:
        return {"status": "success"}
    else:
        return {"status": "failed", "attempts": state["attempts"] + 1}

workflow.add_node("select_card", select_card)
workflow.add_node("process_payment", process_payment)

# ‡πÄ‡∏ä‡∏∑‡πà‡∏≠‡∏°‡∏ï‡πà‡∏≠ nodes
workflow.add_edge("analyze_gateway", "select_card")
workflow.add_edge("select_card", "process_payment")

        return "end"
    elif state["attempts"] >= 3:
        return "end"
    else:
        return "retry"

workflow.add_conditional_edges(
    "process_payment",
    should_retry,
    {
        "end": END,
        "retry": "select_card"
    }
)

    "attempts": 0,
    "status": "start",
    "card_bin": ""
})
```
